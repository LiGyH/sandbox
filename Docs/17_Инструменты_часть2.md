## Шаг 17.1: Balloon — Воздушный шарик

Шарик — это физический объект с **отрицательной гравитацией** (летит вверх). Привязывается к объекту верёвкой.

Создай файл `Code/Weapons/ToolGun/Modes/Balloon.cs`:

```csharp
/// <summary>
/// Создаёт воздушный шарик, привязанный к объекту.
/// Шарик тянет объект вверх.
/// </summary>
[Title( "Balloon" ), Description( "Attach floating balloons" )]
[Icon( "celebration" )]
public class Balloon : ToolMode
{
    [Property] public float BuoyancyForce { get; set; } = 200f;
    [Property] public Color BalloonColor { get; set; } = Color.Red;
    [Property] public string BalloonModel { get; set; } = "models/balloon/balloon.vmdl";

    public override void OnPrimaryAttack()
    {
        if ( !Networking.IsHost ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        // Создаём шарик над точкой попадания
        var balloonPos = tr.HitPosition + Vector3.Up * 64f;
        var balloon = new GameObject( true, "Balloon" );
        balloon.WorldPosition = balloonPos;

        // Модель
        var renderer = balloon.AddComponent<ModelRenderer>();
        renderer.Model = Model.Load( BalloonModel );
        renderer.Tint = BalloonColor;

        // Физика (летит вверх)
        var body = balloon.AddComponent<Rigidbody>();
        body.Gravity = false; // Нет гравитации

        var collider = balloon.AddComponent<SphereCollider>();
        collider.Radius = 16f;

        // Компонент плавучести — толкает вверх каждый кадр
        var buoyancy = balloon.AddComponent<BalloonForce>();
        buoyancy.Force = BuoyancyForce;

        // Привязываем к объекту (если есть Rigidbody)
        var targetBody = tr.GameObject?.GetComponent<Rigidbody>();
        if ( targetBody.IsValid() )
        {
            var joint = balloon.AddComponent<SpringJoint>();
            joint.Body = targetBody;
            joint.MaxLength = 100f;
        }

        balloon.NetworkSpawn();
        Ownable.Set( balloon, Player.Network.Owner );

        AddToUndo( "Balloon", balloon );
    }
}
```

Создай файл `Code/Components/BalloonForce.cs`:

```csharp
/// <summary>
/// Толкает объект вверх каждый кадр (эффект плавучести).
/// </summary>
public class BalloonForce : Component
{
    [Property] public float Force { get; set; } = 200f;

    protected override void OnFixedUpdate()
    {
        var body = GetComponent<Rigidbody>();
        if ( !body.IsValid() ) return;

        // Применяем силу вверх
        body.PhysicsBody.ApplyForce( Vector3.Up * Force );
    }
}
```

### Физика шарика:

```
           ↑ Сила вверх (BalloonForce)
           │
     ┌─────●─────┐
     │   Шарик   │
     └─────┼─────┘
           │ Верёвка (SpringJoint)
           │
     ┌─────●─────┐
     │   Объект  │
     └───────────┘
           │
           ↓ Гравитация

Если шариков достаточно: сила вверх > гравитация → объект взлетает!
```

## Шаг 17.2: Thruster — Двигатель

Создай файл `Code/Weapons/ToolGun/Modes/Thruster.cs`:

```csharp
/// <summary>
/// Прикрепляет двигатель к объекту. Толкает в одном направлении.
/// </summary>
[Title( "Thruster" ), Description( "Attach a thruster" )]
[Icon( "flight" )]
public class Thruster : ToolMode
{
    [Property] public float ThrusterForce { get; set; } = 500f;

    public override void OnPrimaryAttack()
    {
        if ( !Networking.IsHost ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var targetBody = tr.GameObject?.GetComponent<Rigidbody>();
        if ( !targetBody.IsValid() ) return;

        // Создаём двигатель на поверхности объекта
        var thruster = new GameObject( true, "Thruster" );
        thruster.WorldPosition = tr.HitPosition;
        thruster.WorldRotation = Rotation.LookAt( tr.Normal );
        thruster.Parent = tr.GameObject;

        // Компонент двигателя
        var engine = thruster.AddComponent<ThrusterEngine>();
        engine.Force = ThrusterForce;
        engine.TargetBody = targetBody;
        engine.Direction = tr.Normal; // Толкает ОТ поверхности

        thruster.NetworkSpawn();
        AddToUndo( "Thruster", thruster );
    }
}
```

Создай файл `Code/Components/ThrusterEngine.cs`:

```csharp
/// <summary>
/// Двигатель — толкает Rigidbody в заданном направлении.
/// Активируется нажатием определённой клавиши.
/// </summary>
public class ThrusterEngine : Component
{
    [Property] public float Force { get; set; } = 500f;
    [Property] public Vector3 Direction { get; set; } = Vector3.Forward;
    [Property] public Rigidbody TargetBody { get; set; }

    /// <summary>
    /// Активен ли двигатель. По умолчанию всегда включён.
    /// </summary>
    [Sync] public bool IsActive { get; set; } = true;

    protected override void OnFixedUpdate()
    {
        if ( !IsActive ) return;
        if ( !TargetBody.IsValid() ) return;

        // Толкаем в направлении двигателя (с учётом поворота)
        var worldDirection = WorldRotation * Direction;
        TargetBody.PhysicsBody.ApplyForceAt( 
            worldDirection * Force, WorldPosition );
    }
}
```

/// <summary>
/// Создаёт колесо (вращающийся шарнир) на объекте.
/// </summary>
[Title( "Wheel" ), Description( "Add a wheel to an object" )]
[Icon( "settings" )]
public class Wheel : ToolMode
{
    public override void OnPrimaryAttack()
    {
        if ( !Networking.IsHost ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var targetBody = tr.GameObject?.GetComponent<Rigidbody>();
        if ( !targetBody.IsValid() ) return;

        // Создаём колесо
        var wheel = new GameObject( true, "Wheel" );
        wheel.WorldPosition = tr.HitPosition;
        wheel.WorldRotation = Rotation.LookAt( tr.Normal );

        // Физическое тело колеса
        var body = wheel.AddComponent<Rigidbody>();
        var collider = wheel.AddComponent<SphereCollider>();
        collider.Radius = 16f;

        // Шарнир (HingeJoint) — позволяет вращение
        var hinge = wheel.AddComponent<HingeJoint>();
        hinge.Body = targetBody;
        hinge.Axis = tr.Normal; // Ось вращения

        wheel.NetworkSpawn();
        AddToUndo( "Wheel", wheel );
    }
}

## Шаг 17.4: Light — Источник света

Создай файл `Code/Weapons/ToolGun/Modes/LightTool.cs`:

```csharp
/// <summary>
/// Создаёт источник света в указанной точке.
/// </summary>
[Title( "Light" ), Description( "Place a light" )]
[Icon( "lightbulb" )]
public class LightTool : ToolMode
{
    [Property] public Color LightColor { get; set; } = Color.White;
    [Property] public float LightRadius { get; set; } = 500f;
    [Property] public float Brightness { get; set; } = 2f;

    public override void OnPrimaryAttack()
    {
        if ( !Networking.IsHost ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        // Создаём свет
        var lightObj = new GameObject( true, "Light" );
        lightObj.WorldPosition = tr.HitPosition + tr.Normal * 16f;

        // Компонент точечного света
        var light = lightObj.AddComponent<PointLight>();
        light.LightColor = LightColor;
        light.Radius = LightRadius;
        light.Brightness = Brightness;

        // Если есть объект — привязываем
        if ( tr.GameObject.IsValid() )
        {
            lightObj.Parent = tr.GameObject;
        }

        lightObj.NetworkSpawn();
        AddToUndo( "Light", lightObj );
    }
}
```

## Шаг 17.5: Color — Перекраска

Создай файл `Code/Weapons/ToolGun/Modes/ColorTool.cs`:

```csharp
/// <summary>
/// Меняет цвет объекта.
/// </summary>
[Title( "Color" ), Description( "Paint objects" )]
[Icon( "palette" )]
public class ColorTool : ToolMode
{
    [Property] public Color SelectedColor { get; set; } = Color.Red;

    public override void OnPrimaryAttack()
    {
        ApplyColor( SelectedColor );
    }

    [Rpc.Host]
    private void ApplyColor( Color color )
    {
        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var renderer = tr.GameObject?.GetComponent<ModelRenderer>();
        if ( !renderer.IsValid() ) return;

        renderer.Tint = color;
        
        // Обновляем по сети
        tr.GameObject.Network?.Refresh();
    }
}
```

## Шаг 17.6: Duplicator — Копирование

Дупликатор — мощный инструмент, который **сохраняет** сложную конструкцию и **вставляет** её копию.

Создай файл `Code/Weapons/ToolGun/Modes/Duplicator.cs`:

```csharp
/// <summary>
/// Копирует конструкции. ПКМ = копировать, ЛКМ = вставить.
/// </summary>
[Title( "Duplicator" ), Description( "Copy and paste constructions" )]
[Icon( "content_copy" )]
public class Duplicator : ToolMode
{
    private string _copiedData;

    /// <summary>
    /// ПКМ — копировать объект и все связанные.
    /// </summary>
    public override void OnSecondaryAttack()
    {
        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var root = tr.GameObject?.FindNetworkRoot() ?? tr.GameObject;
        if ( !root.IsValid() ) return;

        // Сериализуем объект в JSON
        _copiedData = Scene.Serialize( root ).ToJsonString();
        
        Log.Info( $"Copied: {root.Name}" );
    }

    /// <summary>
    /// ЛКМ — вставить копию.
    /// </summary>
    public override void OnPrimaryAttack()
    {
        Paste();
    }

    [Rpc.Host]
    private void Paste()
    {
        if ( string.IsNullOrEmpty( _copiedData ) ) return;

        var player = Player;
        if ( !player.IsValid() ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        // Десериализуем из JSON
        var json = System.Text.Json.JsonDocument.Parse( _copiedData );
        var go = Scene.Deserialize( json );
        
        if ( go.IsValid() )
        {
            go.WorldPosition = tr.EndPosition;
            go.NetworkSpawn();
            Ownable.Set( go, player.Network.Owner );

            AddToUndo( "Duplicate", go );
        }
    }
}
```
