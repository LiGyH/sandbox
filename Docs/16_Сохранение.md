# Этап 16: Сохранение и загрузка

## Цель этого этапа

Создать систему сохранения и загрузки игрового мира. Игрок может сохранить все созданные объекты и загрузить их позже.

## Как работает сохранение

Сохранение — это **снимок** всех объектов в сцене: их позиция, поворот, свойства.

```
Сохранение:
Сцена → Сериализация → JSON → Файл на диске

Загрузка:
Файл на диске → JSON → Десериализация → Восстановление сцены
```

## Шаг 16.1: ISaveEvents — интерфейс для участников сохранения

Компоненты, которые хотят участвовать в сохранении/загрузке, реализуют `ISaveEvents`:

Создай файл `Code/Save/ISaveEvents.cs`:

```csharp
/// <summary>
/// Интерфейс для компонентов, участвующих в сохранении/загрузке.
/// </summary>
public interface ISaveEvents : ISceneEvent<ISaveEvents>
{
    /// <summary>
    /// Вызывается ПЕРЕД сохранением. Компонент может подготовить данные.
    /// </summary>
    void BeforeSave( string filename ) { }

    /// <summary>
    /// Вызывается ПОСЛЕ загрузки. Компонент может восстановить состояние.
    /// </summary>
    void AfterLoad( string filename ) { }
}
```

## Шаг 16.2: SaveSystem — основная система

Создай файл `Code/Save/SaveSystem.cs`:

```csharp
/// <summary>
/// Система сохранения и загрузки мира.
/// Сохраняет все сетевые объекты и их свойства.
/// </summary>
public class SaveSystem : GameObjectSystem<SaveSystem>
{
    public SaveSystem( Scene scene ) : base( scene ) { }

    public static SaveSystem Current => 
        Game.ActiveScene?.GetSystem<SaveSystem>();

    /// <summary>
    /// Метаданные сохранения (дополнительная информация).
    /// </summary>
    private Dictionary<string, string> _metadata = new();

    public void SetMetadata( string key, string value )
    {
        _metadata[key] = value;
    }

    public string GetMetadata( string key )
    {
        return _metadata.TryGetValue( key, out var value ) ? value : null;
    }

    // --- Сохранение ---

    /// <summary>
    /// Сохранить текущее состояние мира.
    /// </summary>
    public async Task Save( string filename )
    {
        if ( !Networking.IsHost ) return;

        // 1. Уведомляем все компоненты о предстоящем сохранении
        ISaveEvents.Post( x => x.BeforeSave( filename ) );

        // 2. Собираем данные всех объектов
        var saveData = new SaveData
        {
            Timestamp = DateTime.Now,
            MapName = GetCurrentMapName(),
            Objects = CollectObjects(),
            Metadata = new Dictionary<string, string>( _metadata )
        };

        // 3. Сериализуем в JSON
        var json = Json.Serialize( saveData );

        // 4. Сохраняем в файл
        await SaveToFile( filename, json );

        Log.Info( $"Saved: {filename} ({saveData.Objects.Count} objects)" );
    }

    /// <summary>
    /// Собирает все объекты для сохранения.
    /// </summary>
    private List<SavedObject> CollectObjects()
    {
        var objects = new List<SavedObject>();

        foreach ( var go in Scene.GetAllObjects( false ) )
        {
            // Пропускаем объекты без сети
            if ( go.NetworkMode == NetworkMode.Never ) continue;
            
            // Пропускаем игроков (они респавнятся)
            if ( go.Tags.Has( "player" ) ) continue;
            
            // Пропускаем временные объекты
            if ( go.Flags.HasFlag( GameObjectFlags.NotSaved ) ) continue;

            var saved = new SavedObject
            {
                Name = go.Name,
                PrefabSource = go.PrefabInstanceSource,
                Position = go.WorldPosition,
                Rotation = go.WorldRotation,
                Scale = go.WorldScale,
                Tags = go.Tags.ToArray()
            };

            // Сохраняем свойства компонентов
            saved.Components = CollectComponents( go );

            objects.Add( saved );
        }

        return objects;
    }

    /// <summary>
    /// Собирает свойства компонентов объекта.
    /// </summary>
    private List<SavedComponent> CollectComponents( GameObject go )
    {
        var components = new List<SavedComponent>();

        foreach ( var component in go.Components.GetAll() )
        {
            var saved = new SavedComponent
            {
                TypeName = component.GetType().FullName,
                Properties = new Dictionary<string, object>()
            };

            // Сохраняем свойства с атрибутом [Property]
            var type = TypeLibrary.GetType( component.GetType() );
            if ( type != null )
            {
                foreach ( var prop in type.Properties
                    .Where( p => p.HasAttribute<PropertyAttribute>() ) )
                {
                    saved.Properties[prop.Name] = prop.GetValue( component );
                }
            }

            components.Add( saved );
        }

        return components;
    }

    // --- Загрузка ---

    /// <summary>
    /// Загрузить сохранение.
    /// </summary>
    public async Task Load( string filename )
    {
        if ( !Networking.IsHost ) return;

        // 1. Читаем файл
        var json = await LoadFromFile( filename );
        if ( string.IsNullOrEmpty( json ) ) return;

        // 2. Десериализуем
        var saveData = Json.Deserialize<SaveData>( json );
        if ( saveData == null ) return;

        // 3. Очищаем текущую сцену (кроме игроков и системных объектов)
        ClearScene();

        // 4. Восстанавливаем метаданные
        _metadata = new Dictionary<string, string>( saveData.Metadata ?? new() );

        // 5. Восстанавливаем объекты
        foreach ( var saved in saveData.Objects )
        {
            RestoreObject( saved );
        }

        // 6. Уведомляем компоненты о загрузке
        ISaveEvents.Post( x => x.AfterLoad( filename ) );

        Log.Info( $"Loaded: {filename} ({saveData.Objects.Count} objects)" );
    }

    /// <summary>
    /// Удаляет все объекты, созданные игроками.
    /// </summary>
    private void ClearScene()
    {
        foreach ( var go in Scene.GetAllObjects( false ).ToArray() )
        {
            if ( go.Tags.Has( "player" ) ) continue;
            if ( go.Flags.HasFlag( GameObjectFlags.NotSaved ) ) continue;
            if ( go.NetworkMode == NetworkMode.Never ) continue;

            go.Destroy();
        }
    }

    /// <summary>
    /// Восстанавливает один объект из сохранения.
    /// </summary>
    private void RestoreObject( SavedObject saved )
    {
        GameObject go;

        // Если есть источник-префаб — клонируем его
        if ( !string.IsNullOrEmpty( saved.PrefabSource ) )
        {
            var prefab = GameObject.GetPrefab( saved.PrefabSource );
            if ( prefab == null ) return;

            go = prefab.Clone( new CloneConfig
            {
                Transform = new Transform( saved.Position, saved.Rotation, saved.Scale.x ),
                StartEnabled = true
            });
        }
        else
        {
            go = new GameObject( true, saved.Name );
            go.WorldPosition = saved.Position;
            go.WorldRotation = saved.Rotation;
            go.WorldScale = saved.Scale;
        }

        go.NetworkSpawn();
    }

    // --- Вспомогательные ---

    private string GetCurrentMapName()
    {
        return Scene.GetAllComponents<MapInstance>()
            .FirstOrDefault()?.MapName ?? "unknown";
    }

    private async Task SaveToFile( string filename, string json )
    {
        // Используем Storage API движка
        var storage = Storage.GetOrCreate( "saves" );
        await storage.Files.WriteAllTextAsync( $"/{filename}.json", json );
    }

    private async Task<string> LoadFromFile( string filename )
    {
        var storage = Storage.GetOrCreate( "saves" );
        return await storage.Files.ReadAllTextAsync( $"/{filename}.json" );
    }
}
```

## Шаг 16.3: Структуры данных сохранения

Создай файл `Code/Save/SaveData.cs`:

```csharp
/// <summary>
/// Корневая структура файла сохранения.
/// </summary>
public class SaveData
{
    public DateTime Timestamp { get; set; }
    public string MapName { get; set; }
    public List<SavedObject> Objects { get; set; }
    public Dictionary<string, string> Metadata { get; set; }
}

/// <summary>
/// Один сохранённый объект.
/// </summary>
public class SavedObject
{
    public string Name { get; set; }
    public string PrefabSource { get; set; }
    public Vector3 Position { get; set; }
    public Rotation Rotation { get; set; }
    public Vector3 Scale { get; set; }
    public string[] Tags { get; set; }
    public List<SavedComponent> Components { get; set; }
}

/// <summary>
/// Один сохранённый компонент.
/// </summary>
public class SavedComponent
{
    public string TypeName { get; set; }
    public Dictionary<string, object> Properties { get; set; }
}
```
