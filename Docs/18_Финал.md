## Шаг 18.1: Undo система (Отмена)

Система отмены позволяет нажать **Z** чтобы удалить последний созданный объект.

Создай файл `Code/Utility/UndoSystem.cs`:

```csharp
/// <summary>
/// Система отмены (Undo). Каждый игрок имеет свой стек отмены.
/// Нажатие Z удаляет последний созданный объект.
/// </summary>
public class UndoSystem
{
    private Stack<UndoEntry> _entries = new();

    /// <summary>
    /// Создаёт новую запись отмены.
    /// </summary>
    public UndoEntry Create()
    {
        var entry = new UndoEntry();
        _entries.Push( entry );
        return entry;
    }

    /// <summary>
    /// Отменяет последнее действие.
    /// </summary>
    public bool Undo()
    {
        if ( _entries.Count == 0 ) return false;

        var entry = _entries.Pop();
        entry.Execute();
        return true;
    }
}

/// <summary>
/// Одна запись отмены — список объектов для удаления.
/// </summary>
public class UndoEntry
{
    public string Name { get; set; }
    private List<GameObject> _objects = new();

    public void Add( GameObject go )
    {
        _objects.Add( go );
    }

    public void Execute()
    {
        foreach ( var go in _objects )
        {
            if ( go.IsValid() )
                go.Destroy();
        }
        _objects.Clear();
    }
}
```

### Как работает Undo:

```
Игрок спавнит бочку → UndoEntry("Spawn barrel") → [barrel]
Игрок спавнит стул  → UndoEntry("Spawn chair")  → [chair]
Игрок нажимает Z    → Удаляет стул
Игрок нажимает Z    → Удаляет бочку
```

## Шаг 18.2: Система очистки (Cleanup)

Создай файл `Code/Cleanup/CleanupSystem.cs`:

```csharp
/// <summary>
/// Очищает карту от объектов, созданных игроками.
/// Вызывается командой или автоматически.
/// </summary>
public class CleanupSystem : GameObjectSystem<CleanupSystem>
{
    public CleanupSystem( Scene scene ) : base( scene ) { }

    /// <summary>
    /// Очистить все объекты, созданные игроками.
    /// </summary>
    [ConCmd( "cleanup" )]
    public static void CleanupAll()
    {
        if ( !Networking.IsHost ) return;

        var current = Game.ActiveScene?.GetSystem<CleanupSystem>();
        current?.DoCleanup();
    }

    private void DoCleanup()
    {
        int removed = 0;

        foreach ( var go in Scene.GetAllObjects( false ).ToArray() )
        {
            // Пропускаем игроков
            if ( go.Tags.Has( "player" ) ) continue;
            
            // Пропускаем карту
            if ( go.GetComponent<MapInstance>().IsValid() ) continue;
            
            // Пропускаем системные объекты
            if ( go.Flags.HasFlag( GameObjectFlags.NotSaved ) ) continue;

            // Удаляем если есть сетевой режим (значит создан игроком)
            if ( go.NetworkMode == NetworkMode.Object )
            {
                go.Destroy();
                removed++;
            }
        }

        GameManager.Current?.OnCleanup( removed, 0 );
    }
}
```

## Шаг 18.3: Свободная камера (FreeCam)

Создай файл `Code/FreeCam/FreeCamGameObjectSystem.cs`:

```csharp
/// <summary>
/// Свободная камера для наблюдения. Отвязана от игрока.
/// Переключается по нажатию определённой клавиши.
/// </summary>
public class FreeCamGameObjectSystem : GameObjectSystem<FreeCamGameObjectSystem>
{
    public FreeCamGameObjectSystem( Scene scene ) : base( scene ) { }

    public bool IsActive { get; private set; }

    private Vector3 _position;
    private Angles _angles;
    private float _speed = 300f;

    /// <summary>
    /// Переключить свободную камеру.
    /// </summary>
    public void Toggle()
    {
        IsActive = !IsActive;

        if ( IsActive )
        {
            // Запоминаем позицию камеры игрока
            var camera = Scene.Camera;
            if ( camera.IsValid() )
            {
                _position = camera.WorldPosition;
                _angles = camera.WorldRotation.Angles();
            }
        }
    }

    /// <summary>
    /// Обновление свободной камеры.
    /// </summary>
    public void Update()
    {
        if ( !IsActive ) return;

        // Поворот мышью
        _angles += new Angles( -Input.MouseDelta.y * 0.1f, Input.MouseDelta.x * 0.1f, 0 );

        // Движение WASD
        var move = Vector3.Zero;
        if ( Input.Down( "Forward" ) ) move += Vector3.Forward;
        if ( Input.Down( "Backward" ) ) move += Vector3.Backward;
        if ( Input.Down( "Left" ) ) move += Vector3.Left;
        if ( Input.Down( "Right" ) ) move += Vector3.Right;

        var speed = _speed;
        if ( Input.Down( "Run" ) ) speed *= 3f;

        _position += _angles.ToRotation() * move * speed * Time.Delta;

        // Применяем к камере
        var camera = Scene.Camera;
        if ( camera.IsValid() )
        {
            camera.WorldPosition = _position;
            camera.WorldRotation = _angles.ToRotation();
        }
    }
}
```

## Шаг 18.4: Вспомогательные расширения

Создай файл `Code/EngineAdditions.cs`:

```csharp
/// <summary>
/// Расширения для классов движка.
/// </summary>

/// <summary>
/// Расширения для GameObject.
/// </summary>
extension( GameObject go )
{
    /// <summary>
    /// Находит корневой сетевой объект вверх по иерархии.
    /// </summary>
    public GameObject FindNetworkRoot()
    {
        if ( go.NetworkMode == NetworkMode.Object ) 
            return go;
        return go.Parent?.FindNetworkRoot();
    }
}

/// <summary>
/// Расширения для Player — поиск по соединению.
/// </summary>
extension( Player )
{
    /// <summary>
    /// Находит игрока по его сетевому соединению.
    /// </summary>
    public static Player FindForConnection( Connection connection )
    {
        return Game.ActiveScene.GetAll<Player>()
            .FirstOrDefault( x => x.Network.Owner == connection );
    }
}
```

## Шаг 18.6: Советы по развитию проекта

### Как добавлять новое оружие:
1. Создай класс, наследующий `BaseBulletWeapon` или `MeleeWeapon`
2. Переопредели параметры (урон, скорость, разброс)
3. Создай префаб с компонентами
4. Добавь путь к префабу в `GiveDefaultWeapons()` или меню спавна

### Как добавлять новые инструменты:
1. Создай класс, наследующий `ToolMode`
2. Переопредели `OnPrimaryAttack()`, `OnSecondaryAttack()`
3. Добавь компонент как дочерний объект в префаб Tool Gun

### Как добавлять NPC:
1. Создай класс, наследующий `Npc`
2. Переопредели `CreateDefaultSchedule()`
3. Создай новые расписания и задачи
4. Создай префаб с NavMeshAgent и твоим компонентом
