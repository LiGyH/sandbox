## Шаг 11.1: Rope — Верёвка

Верёвка соединяет два объекта гибким «тросом».

Создай файл `Code/Weapons/ToolGun/Modes/Rope.cs`:

```csharp
/// <summary>
/// Верёвка — соединяет два объекта гибкой связью.
/// Объекты могут раскачиваться, но не разлетаться.
/// </summary>
[Title( "Rope" ), Description( "Connect two objects with a rope" )]
[Icon( "cable" )]
public class Rope : ToolMode
{
    private GameObject _firstObject;
    private Vector3 _firstHitPoint;

    public override void OnPrimaryAttack()
    {
        if ( !Networking.IsHost ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var go = tr.GameObject;
        if ( !go.IsValid() || go.Tags.Has( "player" ) ) return;

        // Первый клик — запоминаем объект
        if ( !_firstObject.IsValid() )
        {
            _firstObject = go;
            _firstHitPoint = tr.HitPosition;
            return;
        }

        // Второй клик — создаём верёвку
        var firstBody = _firstObject.GetComponent<Rigidbody>();
        var secondBody = go.GetComponent<Rigidbody>();

        if ( !firstBody.IsValid() || !secondBody.IsValid() )
        {
            _firstObject = null;
            return;
        }

        // SpringJoint с высокой жёсткостью = верёвка
        var joint = _firstObject.AddComponent<SpringJoint>();
        joint.Body = secondBody;
        joint.LocalAnchor = _firstObject.WorldTransform
            .PointToLocal( _firstHitPoint );
        joint.MaxLength = (_firstHitPoint - tr.HitPosition).Length;
        joint.MinLength = 0;
        
        // Визуальная верёвка (линия между объектами)
        // В полной версии используется LineRenderer

        _firstObject = null;
        Sound.Play( "tool_rope", WorldPosition );
    }

    public override void OnSecondaryAttack()
    {
        _firstObject = null;
    }
}
```

### Разница между типами соединений:

```
Weld (жёсткое):    [A]═════[B]     Не сгибается, не тянется
Rope (верёвка):    [A]~~~~~~[B]     Сгибается, НЕ тянется
Elastic (пружина): [A]≈≈≈≈≈≈[B]     Сгибается, ТЯНЕТСЯ
Slider (ползунок): [A]──────[B]     Только по одной оси
BallSocket:        [A]●─────[B]     Вращение в точке крепления
```

## Шаг 11.3: Resizer — Изменение размера

Создай файл `Code/Weapons/ToolGun/Modes/Resizer.cs`:

```csharp
/// <summary>
/// Изменение размера объекта.
/// ЛКМ — увеличить, ПКМ — уменьшить.
/// </summary>
[Title( "Resizer" ), Description( "Resize objects" )]
[Icon( "photo_size_select_large" )]
public class Resizer : ToolMode
{
    [Property] public float ScaleStep { get; set; } = 0.1f;
    [Property] public float MinScale { get; set; } = 0.1f;
    [Property] public float MaxScale { get; set; } = 10f;

    public override void OnPrimaryAttack()
    {
        Resize( ScaleStep ); // Увеличить
    }

    public override void OnSecondaryAttack()
    {
        Resize( -ScaleStep ); // Уменьшить
    }

    public override void OnReload()
    {
        ResetSize(); // Сбросить размер
    }

    [Rpc.Host]
    private void Resize( float delta )
    {
        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var go = tr.GameObject;
        if ( !go.IsValid() || go.Tags.Has( "player" ) ) return;

        var root = go.FindNetworkRoot() ?? go;
        var currentScale = root.WorldScale.x;
        var newScale = Math.Clamp( currentScale + delta, MinScale, MaxScale );

        root.WorldScale = new Vector3( newScale );
    }

    [Rpc.Host]
    private void ResetSize()
    {
        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var go = tr.GameObject;
        if ( !go.IsValid() ) return;

        var root = go.FindNetworkRoot() ?? go;
        root.WorldScale = Vector3.One; // Размер 1:1
    }
}
```

## Шаг 11.4: Mass — Изменение массы

Создай файл `Code/Weapons/ToolGun/Modes/Mass.cs`:

```csharp
/// <summary>
/// Изменение массы объекта.
/// Влияет на то, насколько "тяжёлым" будет объект в физике.
/// </summary>
[Title( "Mass" ), Description( "Change the mass of objects" )]
[Icon( "fitness_center" )]
public class Mass : ToolMode
{
    [Property] public float MassMultiplier { get; set; } = 2f;

    public override void OnPrimaryAttack()
    {
        ChangeMass( MassMultiplier ); // Увеличить массу
    }

    public override void OnSecondaryAttack()
    {
        ChangeMass( 1f / MassMultiplier ); // Уменьшить
    }

    [Rpc.Host]
    private void ChangeMass( float multiplier )
    {
        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var body = tr.GameObject?.GetComponent<Rigidbody>();
        if ( !body.IsValid() ) return;

        body.PhysicsBody.Mass *= multiplier;
        body.PhysicsBody.Mass = Math.Clamp( body.PhysicsBody.Mass, 1f, 100000f );
    }
}
```

## Шаг 11.5: NoCollide — Отключение столкновений

Создай файл `Code/Weapons/ToolGun/Modes/NoCollide.cs`:

```csharp
/// <summary>
/// Отключает столкновения между двумя объектами.
/// Они будут проходить друг сквозь друга.
/// </summary>
[Title( "No Collide" ), Description( "Disable collisions between objects" )]
[Icon( "layers_clear" )]
public class NoCollide : ToolMode
{
    private GameObject _firstObject;

    public override void OnPrimaryAttack()
    {
        if ( !Networking.IsHost ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var go = tr.GameObject;
        if ( !go.IsValid() ) return;

        if ( !_firstObject.IsValid() )
        {
            _firstObject = go;
            return;
        }

        // Отключаем столкновения между объектами
        var firstCollider = _firstObject.GetComponent<Collider>();
        var secondCollider = go.GetComponent<Collider>();

        if ( firstCollider.IsValid() && secondCollider.IsValid() )
        {
            // Используем PhysicsBody.FilterCollisionWith для отключения
            var firstBody = _firstObject.GetComponent<Rigidbody>();
            var secondBody = go.GetComponent<Rigidbody>();

            if ( firstBody.IsValid() && secondBody.IsValid() )
            {
                firstBody.PhysicsBody.FilterCollisionsWith( 
                    secondBody.PhysicsBody, false );
            }
        }

        _firstObject = null;
    }
}
```

## Шаг 11.6: Trail — Создание следа

Создай файл `Code/Weapons/ToolGun/Modes/Trail.cs`:

```csharp
/// <summary>
/// Добавляет визуальный след к объекту.
/// Когда объект двигается, за ним остаётся цветная линия.
/// </summary>
[Title( "Trail" ), Description( "Add a trail to an object" )]
[Icon( "timeline" )]
public class Trail : ToolMode
{
    [Property] public Color TrailColor { get; set; } = Color.White;
    [Property] public float TrailWidth { get; set; } = 4f;
    [Property] public float TrailLifetime { get; set; } = 2f;

    public override void OnPrimaryAttack()
    {
        if ( !Networking.IsHost ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var go = tr.GameObject;
        if ( !go.IsValid() ) return;

        // Добавляем компонент следа
        var trail = go.GetOrAddComponent<TrailRenderer>();
        trail.Color = new ColorGradient( TrailColor );
        trail.Width = TrailWidth;
        trail.Lifetime = TrailLifetime;
    }

    /// <summary>
    /// ПКМ — удалить след.
    /// </summary>
    public override void OnSecondaryAttack()
    {
        if ( !Networking.IsHost ) return;

        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var trail = tr.GameObject?.GetComponent<TrailRenderer>();
        if ( trail.IsValid() )
        {
            trail.Destroy();
        }
    }
}
```

## Шаг 11.7: Unbreakable — Неразрушимость

Создай файл `Code/Weapons/ToolGun/Modes/Unbreakable.cs`:

```csharp
/// <summary>
/// Делает объект неразрушимым. ЛКМ — включить, ПКМ — выключить.
/// </summary>
[Title( "Unbreakable" ), Description( "Make objects indestructible" )]
[Icon( "shield" )]
public class Unbreakable : ToolMode
{
    public override void OnPrimaryAttack()
    {
        SetUnbreakable( true );
    }

    public override void OnSecondaryAttack()
    {
        SetUnbreakable( false );
    }

    [Rpc.Host]
    private void SetUnbreakable( bool unbreakable )
    {
        var tr = DoTrace();
        if ( !tr.Hit ) return;

        var prop = tr.GameObject?.GetComponent<Prop>();
        if ( !prop.IsValid() ) return;

        prop.Invincible = unbreakable;
    }
}
```

## ✅ Проверка

Каждый инструмент можно проверить отдельно:
- **Rope** — соедини два пропа, они будут висеть на верёвке
- **Elastic** — соедини два пропа, они будут пружинить
- **Resizer** — сделай проп огромным или крошечным
- **Mass** — сделай проп тяжёлым (падает медленнее) или лёгким
- **NoCollide** — два пропа проходят друг сквозь друга
- **Unbreakable** — проп не ломается при ударах
